import requests
from bs4 import BeautifulSoup

要爬取的網址

url = “https://weather.com/en-IN/weather/tenday/l/INKA0344:1:IN“

取得網頁內容

response = requests.get(url)

解析 HTML 內容

soup = BeautifulSoup(response.content, “html.parser”)

找到天氣預報的表格

table = soup.find(“table”, {“class”: “twc-table”})

逐一擷取資料

for row in table.find_all(“tr”):
cells = row.find_all(“td”)
if len(cells) > 0:
day = cells[0].text.strip()
date = cells[1].text.strip()
description = cells[2].text.strip()
temperature = cells[3].text.strip()
precipitation = cells[4].text.strip()
wind = cells[5].text.strip()
humidity = cells[6].text.strip()

    print(f"日期：{date} ({day})")
    print(f"天氣狀況：{description}")
    print(f"氣溫：{temperature}")
    print(f"降雨機率：{precipitation}")
    print(f"風速：{wind}")
    print(f"濕度：{humidity}")
    print("-" * 30)

import requests
from bs4 import BeautifulSoup

要爬取的網址

url = “https://example.com“ # 替換成你想要爬取的網址

取得網頁內容

response = requests.get(url)

解析 HTML 內容

soup = BeautifulSoup(response.content, “html.parser”)

篩選出特定標籤或屬性

例如，找到所有 標籤

all_links = soup.find_all(“a”)
for link in all_links:
print(link.get(“href”))

或者，篩選出特定標籤且具有特定屬性值

例如，找到所有 class 為 “content” 的
標籤

contentdivs = soup.find_all(“div”, class=”content”)
for div in content_divs:
print(div.text)
